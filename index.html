def squares(opp, adj):
    """Calculate the squares of two values

    Args:
        opp (float): Opposite side of a right angled triangle
        adj (float): Adjacent side of right angled triangle

    Returns:
        float, float: The squared inputs
    """
    opp_sq = opp**2
    adj_sq = adj**2
    return opp_sq, adj_sq
